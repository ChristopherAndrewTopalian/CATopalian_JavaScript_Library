# Arithmetic 

[isPrime(n)](isPrime.js)  
isPrime(5), returns, true  

[gcd(a, b)](gcd.js)

gcd(5, 10), returns, 5

[lcm(n)](lcm.js)

[factorial(n)](factorial.js)

[isEven(n)](isEven.js)  
isEven(5), returns, false

[isOdd(n)](isOdd.js)  
isOdd(5), returns, true
